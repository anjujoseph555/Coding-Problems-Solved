#include <iostream>
using namespace std;

int main()
{
    cout << " =====  Program to print a Half Pyramid using * ===== \n\n";
    cout << "Enter the number of rows in the pyramid: ";
    cin >> rows;
    cout << "\n\nThe required pyramid pattern containing " << rows << " rows is:\n\n";
    for (i = 1; i <= rows; i++)
    {
        for (j = 1; j <= i; j++)
        {
            cout << "* ";
        }
        cout << endl;
    }

    cout << "\n\n";
    return 0;
}
